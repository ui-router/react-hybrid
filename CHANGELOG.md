class SomeComponent extends React.Component {

```jsx
class SomeComponent extends React.Component {
  render () {
    // access router via props
    const router = this.props.router;
    // do whatever needed with the router
  }
}

<UIRouterConsumer>
  {router => <SomeComponent router={router} />}
